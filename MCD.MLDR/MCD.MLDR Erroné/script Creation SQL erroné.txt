   MOTDEPASSE CHAR(32) NOT NULL  ,

# -----------------------------------------------------------------------------
#       TABLE : MOTIF
# -----------------------------------------------------------------------------

CREATE TABLE IF NOT EXISTS MOTIF
 (
   NOM VARCHAR(128) NOT NULL  ,
   CONSIGNE CHAR(255) DEFAULT 'Pas de consignes particulières.' ,
   PIECES CHAR(255) NOT NULL  ,
   PRIX DECIMAL(10,2) NOT NULL  
   , PRIMARY KEY (NOM) 
 )

# -----------------------------------------------------------------------------
#       TABLE : PATIENT
